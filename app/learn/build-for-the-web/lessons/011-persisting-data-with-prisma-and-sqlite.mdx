    	store.messages = store.messages.filter(m => m.id !== id)

    		return { ok: false, error: 'Message not found.' }

    }

### Current persistence boundary

- The following file is the **only place where persistence currently lives**:

```ts
import fs from 'fs/promises'
import path from 'path'

const STORE_PATH = path.join(process.cwd(), 'app/data/store.json')

export type Store = {
  messages: {
    id: string
    text: string
    createdAt: number
  }[]
  profile: {
    id: string
    isFollowing: boolean
  }
}

export async function readStore(): Promise<Store> {
  const raw = await fs.readFile(STORE_PATH, 'utf-8')
  return JSON.parse(raw)
}

export async function writeStore(store: Store) {
  await fs.writeFile(STORE_PATH, JSON.stringify(store, null, 2))
}
```

- Everything else in the system depends on this file.
- Nothing else knows **how** data is stored.

That is the boundary we are about to replace.

## Step 2: Introduce a Real Persistence Engine 🧩

- Right now, persistence works.
- But it works because **we implemented it ourselves**.
- Real systems do not:
  - Manually read JSON files.
  - Manually write JSON files.
  - Manually coordinate consistency.
- They delegate persistence to a database.
- And they delegate interaction with that database to a data layer.

> **This step introduces infrastructure — not behavior.**

- The system already knows:
  - What exists.
  - How truth changes.
  - Who owns truth.
- We are not changing any of that.
- We are only replacing **how truth is stored**.

## Step 3: Why SQLite? Why Prisma? 🧠

- Before touching code, we clarify intent.

### SQLite

- A real relational database.
- File-based.
- No server required.
- Deterministic
  > SQLite is deterministic because it has no hidden background
  > behavior—the same inputs always produce the same persisted state.
- Perfect for local systems and learning.
- SQLite is not a toy.
  > It is a database used in production systems worldwide.

### Prisma

- A schema-first persistence tool.
- Explicit data models.
- Explicit migrations.
- Strong typing.
- No hidden magic.

> Prisma is not your system.
> Prisma is **infrastructure**.

- It exists to:
  - Express your data model.
  - Enforce consistency.
  - Generate safe queries.
- The system still decides:
  - What can be created.
  - What can be updated.
  - What can be deleted.

## Step 9: Replace the JSON Store with Prisma

- Replace the code in `app/data/store.server.ts` with the following code:

```ts
import { prisma } from './prisma.server'

export type Store = {
  messages: {
    id: string
    text: string
    createdAt: number
  }[]
  profile: {
    id: string
    isFollowing: boolean
  }
}

const PROFILE_ID = 'profile-1'

export async function readStore(): Promise<Store> {
  const messages = await prisma.message.findMany({
    orderBy: { createdAt: 'asc' },
  })

  let profile = await prisma.profile.findUnique({
    where: { id: PROFILE_ID },
  })

  if (!profile) {
    profile = await prisma.profile.create({
      data: {
        id: PROFILE_ID,
        isFollowing: false,
      },
    })
  }

  return {
    messages,
    profile,
  }
}

export async function writeStore(store: Store) {
  // We no longer write the entire store.
  // Mutations are handled directly via Prisma in actions.
}
```

- Notice:
  - `readStore()` still returns the same shape.
  - `writeStore()` becomes unnecessary.
  - The contract remains intact.

## Step 11: Remove the Old JSON Infrastructure 🧹

- At this point the system is already writing to the database.
- However, some remnants of the JSON persistence layer still remain in
  the codebase.
- These remnants are no longer used and should now be removed.
- Removing them is important because it proves that the persistence swap
  is complete.

### Step 11a: Remove `writeStore` From the Route

- Earlier in Lesson 10, the route imported both helpers:

```ts
import { readStore, writeStore } from '~/data/store.server'
```

- But `writeStore` is no longer needed.
- Replace the import with:

```ts
import { readStore } from '~/data/store.server'
```

- The loader still depends on `readStore`, but mutations now go directly
  through Prisma.

### Step 11b: Remove Unused `store` Variables

- Inside the `action` function you may still see lines like this:

```ts
const store = await readStore()
```

- These were necessary when mutations happened in memory before writing
  to the JSON file.
- Now that Prisma writes directly to the database, these variables are
  no longer used.
- Delete them.
- For example, change this:

```ts
const store = await readStore()

await prisma.message.create({
  data: newMessage,
})
```

- to simply:

```ts
await prisma.message.create({
  data: newMessage,
})
```

- You should remove the unused `store` variable from each intent block.

### Step 11c: Delete the JSON File

- The file:

```text
app/data/store.json
```

- is no longer used.
- Delete it.
- The system no longer reads or writes JSON persistence.

### Step 11d: Confirm the Loader Still Works

- Your route loader **does not change**.
- It should still look like this:

```ts
export async function loader() {
	const store = await readStore()
	return store
}
```

> - Notice that the loader still calls `readStore()` instead of Prisma directly.
> - This preserves the persistence boundary introduced in Lesson 10.

- At first glance this might look strange because we introduced Prisma
  in the previous steps.
- However, remember the boundary we created earlier.
- The loader does **not** know how persistence works.
- It only knows that it can request the current state of the system
  from `readStore()`.
- Originally, `readStore()` loaded data from a JSON file.
- Now `readStore()` loads data from Prisma and SQLite.
- Because the **return shape stayed the same**, the loader does not need
  to change.
- The UI does not change.
- The route does not change.
- The boundary holds.

> The loader still reads the store —
> but the store is now backed by a real database.

- This is the proof that the system was designed correctly.
- Persistence changed completely, yet nothing above the boundary had to change.
